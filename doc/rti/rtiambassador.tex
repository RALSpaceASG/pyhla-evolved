\subsection{RTI Ambassador}

\begin{classdesc}{RTIAmbassador}{}
Provides RTI ambassador services.

\begin{verbatim}
rtia = hla.rti.RTIAmbassador()
rtia.joinFederationExecution("python-01", "MyFederation", fed)
\end{verbatim}
