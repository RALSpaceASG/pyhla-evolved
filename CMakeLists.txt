PROJECT(PyHLA C CXX)
# SET(CMAKE_VERBOSE_MAKEFILE ON)

CMAKE_MINIMUM_REQUIRED(VERSION 2.4)

SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_INSTALL_PREFIX}/share/scripts)
SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_CURRENT_SOURCE_DIR})

FIND_PACKAGE(PythonLibs REQUIRED)
FIND_PACKAGE(PythonInterp REQUIRED)
EXEC_PROGRAM("${PYTHON_EXECUTABLE}"
    ARGS "-c \"from distutils.sysconfig import get_python_lib; print get_python_lib()\""
    OUTPUT_VARIABLE SITE_PACKAGES)
SET(SITE_PACKAGES "${SITE_PACKAGES}/hla")
MESSAGE(STATUS "Python modules will be installed in: ${SITE_PACKAGES}")

ADD_SUBDIRECTORY(hla)
